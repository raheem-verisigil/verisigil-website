from verisigil import RuntimeGuard

# Initialize once
guard = RuntimeGuard(api_key="your-api-key")

# Intercept any agent action before execution
decision = guard.verify(
    agent_id     = "vsa_1483d06a89c4",
    action_type  = "payment",
    action_details = {
        "amount_usd": 50000,
        "recipient": "vendor@acme.com"
    }
)

if decision["decision"] == "ALLOW":
    # Execute the action
    execute_payment(50000)
    
elif decision["decision"] == "REQUIRE_HUMAN_APPROVAL":
    # Agent pauses — approver already emailed
    print(f"Awaiting approval: {decision['approval_url']}")
    
else:  # DENY
    print(f"Blocked: {decision['reason']}")
